00738            for (j=0;j<nComp;j++) PSat[j]=compounds[j]->pSatCorrelation->Value(T);
00739            for (j=0;j<nComp;j++)
00740             {for (k=0;k<nComp;k++)
00741               {//d X[k] / d n[j]
00742                if (k==j) vals[index]=PSat[k]*(1.0-X[k]); else vals[index]=-X[k]*PSat[k];
00743                index++;
00744               }
00745             }
00746           }
00747                  break;   
00748      case FugacityCoefficient:
00749          if (phaseID==Vapor)
00750           {//unity
00751            for (j=0;j<nComp;j++) vals[j]=1.0;
00752           }
00753          else
00754           {//liquid, fug[j]=x[j]*Psat[j]=phi[j]*x[j]*P -> phi[j]=Psat[j]/P
00755            double invP=1.0/P;
00756            for (j=0;j<nComp;j++) vals[j]=compounds[j]->pSatCorrelation->Value(T)*invP;
00757           }
00758                  break;   
00759      case FugacityCoefficientDT:
00760          if (phaseID==Vapor)
00761           {//zero
00762            for (j=0;j<nComp;j++) vals[j]=0.0;
00763           }
00764          else
00765           {//liquid
00766            double invP=1.0/P;
00767            for (j=0;j<nComp;j++) vals[j]=compounds[j]->pSatCorrelation->ValueDT(T)*invP;
00768           }
00769                  break;   
00770      case FugacityCoefficientDP:
00771          if (phaseID==Vapor)
00772           {//zero
00773            for (j=0;j<nComp;j++) vals[j]=0.0;
00774           }
00775          else
00776           {//liquid
00777            double invP2=-1.0/(P*P);
00778            for (j=0;j<nComp;j++) vals[j]=compounds[j]->pSatCorrelation->Value(T)*invP2;
00779           }
00780                  break;   
00781      case FugacityCoefficientDX:
00782      case FugacityCoefficientDn:
00783          //zero for all phases
00784          memset(vals,0,sizeof(double)*nComp*nComp);
00785                  break;   
00786      case LogFugacityCoefficient:
00787          if (phaseID==Vapor)
00788           {//ln(unity)=0
00789            for (j=0;j<nComp;j++) vals[j]=0.0;
00790           }
00791          else
00792           {//liquid, ln(phi[j])=ln(Psat[j]/P)
00793            double lnP=log(P);
00794            for (j=0;j<nComp;j++) vals[j]=log(compounds[j]->pSatCorrelation->Value(T))+lnP;
00795           }
00796                  break;   
00797      case LogFugacityCoefficientDT:
00798          if (phaseID==Vapor)
00799           {//d/dT ln(unity)=0
00800            for (j=0;j<nComp;j++) vals[j]=0.0;
00801           }
00802          else
00803           {//liquid
00804            for (j=0;j<nComp;j++) vals[j]=compounds[j]->pSatCorrelation->ValueDT(T)/compounds[j]->pSatCorrelation->Value(T);
00805           }
00806                  break;   
00807      case LogFugacityCoefficientDP:
00808          if (phaseID==Vapor)
00809           {//d/dP ln(unity)=0
00810            for (j=0;j<nComp;j++) vals[j]=0.0;
00811           }
00812          else
00813           {//liquid
00814            double minInvP=-1.0/P;
00815            for (j=0;j<nComp;j++) vals[j]=minInvP;
00816           }
00817                  break;   
00818      case LogFugacityCoefficientDX:
00819      case LogFugacityCoefficientDn:
00820          memset(vals,0,sizeof(double)*nComp*nComp);
00821                  break;   
00822      case Activity:
00823          if (phaseID==Vapor)
00824           {lastError="Activity not supported for vapor phase";
00825            return false;
00826           }
00827          //liquid activity coefficent is unity, activity therefore equals X
00828          for (j=0;j<nComp;j++) vals[j]=X[j];
00829                  break;   
00830      case ActivityDT:
00831      case ActivityDP:
00832          if (phaseID==Vapor)
00833           {lastError="Activity not supported for vapor phase";
00834            return false;
00835           }
00836          //zero
00837          for (j=0;j<nComp;j++) vals[j]=0;
00838                  break;   
00839      case ActivityDX:
00840          if (phaseID==Vapor)
00841           {lastError="Activity not supported for vapor phase";
00842            return false;
00843           }
00844          //identity matrix
00845          memset(vals,0,sizeof(double)*nComp*nComp);
00846          for (j=0;j<nComp;j++) vals[j+nComp*j]=1.0;
00847                  break;   
00848      case ActivityDn:
00849          if (phaseID==Vapor)
00850           {lastError="Activity not supported for vapor phase";
00851            return false;
00852           }
00853          //for a total of 1 moles:
00854          //d X[i] / d n[i] = 1-X[i]
00855          //d X[i] / d n[j] = -X[i]
00856          index=0;
00857          for (j=0;j<nComp;j++)
00858           {for (k=0;k<nComp;k++)
00859             {//d X[k] / d n[j]
00860              if (k==j) vals[index]=1.0-X[k]; else vals[index]=-X[k];
00861              index++;
00862             }
00863           }
00864                  break;   
00865      default:
00866          lastError="Internal error: property calculation not defined";
00867          return false;
00868     }    
00869   }
00870  //all ok
00871  return true;
00872 }
